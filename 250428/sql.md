```sql
--프로그래머스- 업그레이드 된 아이템 구하기
SELECT A.ITEM_ID, A.ITEM_NAME, A.RARITY
FROM ITEM_INFO A
INNER JOIN ITEM_TREE B ON A.ITEM_ID=B.ITEM_ID
WHERE B.PARENT_ITEM_ID IN (SELECT ITEM_ID FROM ITEM_INFO WHERE RARITY='RARE')
ORDER BY A.ITEM_ID DESC

-- 프로그래머스- 조건에 맞는 개발자 찾기
SELECT ID,EMAIL,FIRST_NAME,LAST_NAME
FROM DEVELOPERS
WHERE SKILL_CODE&(SELECT CODE FROM SKILLCODES WHERE NAME='Python') 
    OR SKILL_CODE&(SELECT CODE FROM SKILLCODES WHERE NAME='C#')
ORDER BY ID

-- 프로그래머스- 부모의 형질을 모두 가지는 대장균 찾기
SELECT A.ID, A.GENOTYPE, B.GENOTYPE AS PARENT_GENOTYPE
FROM ECOLI_DATA A
RIGHT JOIN ECOLI_DATA B ON A.PARENT_ID=B.ID
WHERE A.GENOTYPE & B.GENOTYPE = B.GENOTYPE
```
