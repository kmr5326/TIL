```sql
-- 프로그래머스- 강원도에 위치한 생산공장 목록 출력하기
SELECT FACTORY_ID, FACTORY_NAME, ADDRESS FROM FOOD_FACTORY WHERE ADDRESS LIKE '강원도%';


-- 프로그래머스- 경기도에 위치한 식품창고 목록 출력하기
SELECT WAREHOUSE_ID, WAREHOUSE_NAME, ADDRESS, IFNULL(FREEZER_YN,"N") AS FREEZER_YN 
FROM food_warehouse 
WHERE ADDRESS LIKE '경기도%'
order by warehouse_id
/*NULL처리 함수
IFNULL, COALESCE, NVL ORACLE*/

-- 프로그래머스- 나이 정보가 없는 회원 수 구하기
SELECT COUNT(*) AS USERS FROM USER_INFO WHERE AGE IS NULL

-- 프로그래머스- 조건에 맞는 회원수 구하기
SELECT COUNT(*) FROM USER_INFO
WHERE YEAR(JOINED)='2021' AND AGE BETWEEN 20 AND 29

-- 프로그래머스- 인기있는 아이스크림
SELECT FLAVOR FROM FIRST_HALF
ORDER BY TOTAL_ORDER DESC, SHIPMENT_ID ASC

-- 프로그래머스- 잔챙이 잡은 수 구하기
SELECT COUNT(*) AS FISH_COUNT FROM FISH_INFO
WHERE LENGTH IS NULL

-- 프로그래머스- 특정 옵션이 포함된 자동차 리스트 구하기
SELECT CAR_ID, CAR_TYPE, DAILY_FEE, OPTIONS
FROM CAR_RENTAL_COMPANY_CAR
WHERE OPTIONS LIKE '%네비게이션%'
ORDER BY CAR_ID DESC
-- WHERE INSTR(OPTIONS, '네비게이션') > 0 

-- 프로그래머스- Python 개발자 찾기
SELECT ID, EMAIL, FIRST_NAME, LAST_NAME
FROM DEVELOPER_INFOS
WHERE 'Python' IN (SKILL_1,SKILL_2,SKILL_3)
ORDER BY ID

-- 프로그래머스- 잡은 물고기의 평균 길이 구하기
SELECT ROUND(AVG(COALESCE(LENGTH,10)),2) AS AVERAGE_LENGTH
FROM FISH_INFO

-- 프로그래머스- 자동차 대여 기록에서 장기/단기 대여 구분하기
SELECT HISTORY_ID, CAR_ID, 
DATE_FORMAT(START_DATE,'%Y-%m-%d') AS START_DATE, 
DATE_FORMAT(END_DATE,'%Y-%m-%d') AS END_DATE,
CASE WHEN DATEDIFF(END_DATE,START_DATE)<29 THEN '단기 대여' ELSE '장기 대여' END AS RENT_TYPE
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
WHERE MONTH(START_DATE)=9
ORDER BY HISTORY_ID DESC

-- 프로그래머스- 가장 큰 물고기 10마리 구하기
SELECT ID, LENGTH
FROM FISH_INFO
ORDER BY LENGTH DESC, ID ASC
LIMIT 10

-- 프로그래머스- 조건에 부합하는 중고거래 댓글 조회하기
SELECT A.TITLE, A.BOARD_ID, B.REPLY_ID, B.WRITER_ID, B.CONTENTS, 
DATE_FORMAT(B.CREATED_DATE,'%Y-%m-%d') AS CREATED_DATE
FROM USED_GOODS_BOARD A
INNER JOIN USED_GOODS_REPLY B ON A.BOARD_ID=B.BOARD_ID
WHERE MONTH(A.CREATED_DATE)=10
ORDER BY B.CREATED_DATE ASC, A.TITLE ASC

-- 프로그래머스- 특정 형질을 가지는 대장균 찾기
SELECT COUNT(*) AS COUNT FROM ECOLI_DATA
WHERE GENOTYPE & 2 =0 AND(GENOTYPE & 1 != 0 OR GENOTYPE & 4 != 0)
```
